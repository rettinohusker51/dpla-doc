def branding_img(image_name)
  if defined? DplaFrontendAssets
    case image_name
    when 'logo.png'
      'dpla-logo.png'
    when 'footer-logo.png'
      'dpla-footer-logo.png'
    else
      image_name
    end
  else
    image_name
  end
end

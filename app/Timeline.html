def api_search_path
  conditions = DPLA::Conditions.new({ q: @term }.merge(@filters).merge(fields: fields))
  "#{api_base_path}/items?#{conditions}#{api_key}"
end

def final_year
  year = self.years.max_by{|k,v| k}.first.to_i
  raise unless YEARS.include? year
  year
rescue
  Time.now.year
end
